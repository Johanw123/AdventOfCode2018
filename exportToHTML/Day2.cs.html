10           {

12   
13               Dictionary<char, int> dir = new Dictionary<char, int>();
14   
15               int sum1 = 0;
16               int sum2 = 0;
17               
18               foreach (var line in lines)
19               {
20                   foreach (var letter in line)
21                   {
22                       if (dir.ContainsKey(letter))
23                           dir[letter]++;
24                       else
25                           dir.Add(letter, 1);
26                   }
27   
28                   if (dir.Values.Contains(2))
29                       sum1++;
30                   if (dir.Values.Contains(3))
31                       sum2++;
32                   
33                   dir.Clear();
34               }
35   
36               return (sum1 * sum2).ToString();
37           }

39           public static string RunPart2()
40           {

42   
43               foreach (var line in lines)

45                   foreach (var line2 in lines)

58                               matching++;
59                               result += char1;
60                           } 
61                       }
62   
63                       if (matching == line.Length - 1)
64                           return result;
65                   }                
66               }
